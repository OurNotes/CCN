# <a name="vue.js-02" href="#" >写代码</a>

>1、写json

```json
    "time":"2019-01-02",
    "offset":0,
    "page":1,
    "limit":10,
    "total":84,
    "rows": [
        {
            "test1":"atest1",
            "test2":"atest2"
        },
        {
            "test1":"btest1",
            "test2":"btest2"
        }
    ]
}
```

>2、写html
```js
<div>
  <el-button type="primary" @click="downloadOut">主要按钮</el-button>
  <el-table :data="testTable" style="width: 100%" id="tableData">
    <el-table-column prop="test1" label="测试1" width="180">
    </el-table-column>
    <el-table-column prop="test2" label="测试2" width="180">
    </el-table-column>
  </el-table>
  </div>
```

>3、写js
```js
import XLSX from 'xlsx';
  import FileSaver from 'file-saver'
  export default {
    data() {
      return {
        testTable: []
      }
    },
    created() {
      this.getJson();
    },
    methods: {
      getJson() {
        this.axios.get('static/json/test.json')
          .then((response) => {
            this.testTable = response.data.rows;
          }).catch((response) => {
            console.log(response);
          })
      },
      downloadOut() {
        this.exportExcel();
      },
      exportExcel() { //导出Excel
        let wb = XLSX.utils.table_to_book(document.getElementById('tableData'));
        let wbout = XLSX.write(wb, {
          bookType: 'xlsx',
          type: 'binary'
        });
        FileSaver.saveAs(new Blob([this.s2ab(wbout)], {
          type: 'application/octet-stream'
        }), "tableData.xlsx");
      },
      s2ab: function (s) {
        if (typeof ArrayBuffer !== 'undefined') {
          let buf = new ArrayBuffer(s.length);
          let view = new Uint8Array(buf);
          for (var i = 0; i != s.length; ++i) {
            view[i] = s.charCodeAt(i) & 0xFF;
          }
          return buf;
        } else {
          let buf = new Array(s.length);
          for (var i = 0; i != s.length; ++i) {
            buf[i] = s.charCodeAt(i) & 0xFF;
            return buf;
          }
        }
      }
    }
  }
```
