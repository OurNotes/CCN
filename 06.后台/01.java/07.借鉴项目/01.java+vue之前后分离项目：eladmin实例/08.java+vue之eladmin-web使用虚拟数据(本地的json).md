> 写页面文件index.vue

```html
<template>
  <el-table
    :data="tableData"
    style="width: 100%">
    <el-table-column
      prop="date"
      label="日期"
      width="180"/>
    <el-table-column
      prop="name"
      label="姓名"
      width="180"/>
    <el-table-column
      prop="address"
      label="地址"/>
  </el-table>
</template>

<script>
import axios from 'axios'
export default {
  data() {
    return {
      tableData: []
    }
  },
  mounted: function() {
    this.getJson()
  },
  methods: {
    getJson() {
      axios.get('/static/json/test.json')
        .then((response) => {
          this.tableData = response.data
        }).catch((response) => {
          console.log(response) // 浏览器按F12 进入调试模式，选择Console选项卡，可以看到后台输入
        })
    }
  }

}
</script>
```
